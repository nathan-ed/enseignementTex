\begin{tasks}(2)
	\task $v=\dfrac{d}{t} \hfill d=? \hfill t=?\hfill$ 
		
		Isolons $d:$
			\begin{solve}
				v&=\dfrac{d}{t}&\cdot t\\
				v\cdot t&=d&d \text{ est isolé}
			\end{solve}

			Isolons $t:$
			\begin{solve}
				v&=\dfrac{d}{t}&\cdot t\\
				v\cdot t&=d&\div v\\
				t&=\dfrac{d}{v}&t \text{ est isolé}
			\end{solve}

	\task $P=2(a+b) \hfill b=? \hfill$

			Isolons $b:$
			\begin{solve}
				P&=2(a+b)&\div 2\\
				\dfrac{P}{2}&=a+b&-a\\
				\dfrac{P}{2}-a&=b&b \text{ est isolé}
			\end{solve}

	\task $A=\dfrac{\left(B+b\right)}{2}h \hfill h=? \hfill B=? \hfill$

			Isolons $h$:
			\begin{solve}
				A&=\dfrac{(B+b)}{2}h&\cdot \dfrac{2}{(B+b)}\\
				A\cdot \dfrac{2}{B+b}&=h&\text{réduire}\\
				\dfrac{2A}{B+b}&=h&h \text{ est isolé}
			\end{solve}
		
			Isolons $B$:
			\begin{solve}
			A&=\dfrac{(B+b)}{2}h&\div h\\
			\dfrac{A}{h}&=\dfrac{(B+b)}{2}&\cdot 2\\
			\dfrac{2A}{h}&=B+b&-b\\
			\dfrac{2A}{h}&=B& B \text{ est isolé}
		\end{solve}
	\task $E=mgh \hfill h=? \hfill \phantom{1=?} \hfill$
			\begin{solve}
				E&=mgh&\div mg\\
				\dfrac{E}{mg}&=h&= h \text{ est isolé}
			\end{solve}
	\task $P=f\dfrac{m_1m_2}{m_3} \hfill m_1=? \hfill m_3=? \hfill$
			
		Isolons $m_1$:
			\begin{solve}
				P&=f\dfrac{m_1m_2}{m_3}&\div f\\
				\dfrac{P}{f}&=\dfrac{m_1m_2}{m_3}& \cdot \dfrac{m_3}{m_2}\\
				\dfrac{P}{f}\cdot \dfrac{m_3}{m_2}&=m_1& \text{ réduire}\\
				\dfrac{Pm_3}{fm_2}&=m_1&m_1 \text{ est isolé}
			\end{solve}
		
			Isolons $m_3$ (on reprend la formule ou $m_1$ est isolé):
			\begin{solve}
				\dfrac{Pm_3}{fm_2}&=m_1&\div P\\
				\dfrac{m_3}{fm_2}&=\dfrac{m_1}{P}&\cdot fm_2\\
				m_3&=\dfrac{m_1}{P}\cdot fm_2&\text{réduire}\\
				m_3&=\dfrac{fm_1m_2}{P}&m_3 \text{ est isolé}
			\end{solve}
	\task $\dfrac{n_1}{n_2}=\dfrac{z_1z_4}{z_2z_3} \hfill z_1=? \hfill n_2=? \hfill$
			
		Isolons $z_1$:
			\begin{solve}
			\dfrac{n_1}{n_2}&=\dfrac{z_1z_4}{z_2z_3}&\cdot \dfrac{z_2z_3}{z_4}\\
			\dfrac{n_1}{n_2}\cdot \dfrac{z_2z_3}{z_4}&=z_1&\text{réduire}\\
			\dfrac{n_1z_2z_3}{n_2z_4}&=z_1&z_1 \text{ est isolé}
		\end{solve}
		
		Isolons $n_2$: 
		\begin{solve}
			\dfrac{n_1}{n_2}&=\dfrac{z_1z_4}{z_2z_3}&\cdot n_2\\
			n_1&=\dfrac{z_1z_4}{z_2z_3}\cdot n_2&\cdot \dfrac{z_2z_3}{z_1z_4}\\
			n_1\dfrac{z_2z_3}{z_1z_4}&=n_2&\text{réduire}\\
			\dfrac{n_1z_2z_3}{z_1z_4}&=n_2&n_2 \text{ est isolé}
		\end{solve}
		\task $ a=\dfrac{Ah}{2}-b \hfill h=\dfrac{2A}{a+b} \hfill$
	\task $h=\dfrac{4V}{\pi d^2}$
\end{tasks}
