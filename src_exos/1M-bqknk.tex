\contenu{
\tcblower
\noindent Soient les fonctions \(f\) et \(g\) définies par  
\[
f(x)=x^2+2x+1,\quad g(x)=x^3+2x^2-x-2.
\]
Déterminer les limites suivantes :  
\begin{tasks}(3)
\task \(\displaystyle\lim_{x\to 1}\frac{f(x)}{g(x)}\)
\task \(\displaystyle\lim_{x\to 1}\frac{g(x)}{f(x)}\)
\task \(\displaystyle\lim_{x\to+\infty}\frac{f(x)}{g(x)}\)
\task \(\displaystyle\lim_{x\to+\infty}\frac{g(x)}{f(x)}\)
\task \(\displaystyle\lim_{x\to+\infty}\bigl(f(x)-g(x)\bigr)\)
\task \(\displaystyle\lim_{x\to+\infty}\Bigl(\frac{f(x)}{g(x)}\Bigr)^{3}\)
\end{tasks}
}

		\task $0$
	\end{tasks}
