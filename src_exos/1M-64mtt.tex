	Factoriser le plus possible les expressions suivantes.
	\begin{tasks}(2)
\task $m(a-b)+n(a-b)$
\task $x(2 a-b)+y(b-2 a)$
\task $a(x-y)-(y-x)$
\task $(a+b)(x-3 y)-3 a(x-3 y)$
\task $(a+b)^3-(a+b)^2$
\task $(x-3)(x+1)-x+3+2(x-3)^2$
\task $(a-b)^3-(a-b)$
\task $(x-y)-(a+b)^2(x-y)$
	\end{tasks}
}
