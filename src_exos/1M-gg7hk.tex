\correction{
\tcblower
\begin{tasks}(2)
\task $ \displaystyle \lim_{x \to 0} \dfrac{2x}{x}
= «\dfrac{0}{0}»$

On simplifie: $\dfrac{2x}{x} = 2 \text{ pour } x \ne 0, \lim_{x\to 0} 2 = 2$

\task \( \displaystyle \lim_{x \to 0} \dfrac{5x^2 - 7x}{x}
= «\dfrac{0 - 0}{0}» = «\dfrac{0}{0}» \)
On factorise
\(
\dfrac{x(5x - 7)}{x} = 5x - 7 \Rightarrow \lim_{x\to 0}5x-7 = -7 \)

\task type $«\dfrac{0}{0}»$ 

\(\begin{aligned} \displaystyle \lim_{u \to 2} \dfrac{u^2 - 4}{u^2 - 5u + 6}
&= \lim_{u \to 2}\dfrac{(u - 2)(u + 2)}{(u - 2)(u - 3)}\\
&\stackrel{x\neq 2}{=} \lim_{u \to 2}\dfrac{u + 2}{u - 3}\\
&=\dfrac{4}{-1} = -4 \end{aligned}\)
\task type $«\dfrac{0}{0}»$ 

\(\begin{aligned}
	&\displaystyle \lim_{x \to 1} \dfrac{x^3 + 3x^2 - 2x - 2}{x^2 - 1}\\
	&= \lim_{x\to 1}\dfrac{(x + 1)(x - 1)(x + 2)}{(x - 1)(x + 1)}\\
	&\stackrel{x\neq 1,-1}{=}
\lim_{x\to 1} (x + 2) = 3 \end{aligned}\)

\task type $«\dfrac{0}{0}»$

\( \begin{aligned}
	\displaystyle \lim_{x \to 3} \dfrac{x^3 - 27}{x - 3}
	&= \dfrac{(x - 3)(x^2 + 3x + 9)}{x - 3} \\
	&\stackrel{x\neq 3}{=}\lim_{x\to 3} x^2 + 3x + 9 = 27\end{aligned} \)

\task type $«\dfrac{0}{0}»$ 

\( \displaystyle \lim_{x \to 0} \dfrac{x^3}{x^2}
\stackrel{x\neq 0}{=} \lim_{x\to 0} x = 0 \)
\end{tasks}
