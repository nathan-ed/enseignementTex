\contenu{
	\tcblower
On considère les deux fonctions suivantes : $f: x \mapsto \sqrt{x}$ et $g: x \mapsto \dfrac{1}{x}$.
\begin{tasks}
\task Donner leur domaine de définition.
\task Représenter graphiquement $f$ sur l'intervalle $[0 ; 2]$ en prenant 4 carrés par unité horizontalement et verticalement. (Calculer au moins 5 images entre $0$ et $1$.)
\task Représenter graphiquement $g$ sur l'intervalle $[-5 ; 5]$ en prenant 2 carrés par unité. (Calculer une dizaine d'images entre $-1$ et $1$.)
\end{tasks}
}
\correction{
	\tcblower
$D_f=\R_+$ et $D_g=R^*$.
}
