\begin{tasks}
	\task Représenter $f(x)$ et $g(x)$. 
	\task Résoudre $f(x)=g(x)$.
	\task Déterminer à l'aide des calculs et du graphique les solutions de $f(x)<g(x)$. 
\end{tasks}
