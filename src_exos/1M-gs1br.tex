\contenu{
	\tcblower
Répondre aux questions suivantes pour la fonction $f(x)=\lvert x-2\rvert$~:
\begin{tasks}(2)
	\task Déterminer l'image de $-3$ et $6$.
	\task Déterminer la préimage de $-2$ et de $-2$ par $f$.
	\task Déterminer la préimage de $0$ par $f$.
	\task Déterminer les zéros de $f$.
\end{tasks}
}
\correction{
	\tcblower
\begin{tasks}(3)
	\task $f(-3)=5$ $f(6)=4$
	\task $f^{-1}(-2)=\emptyset$, $f^{-1}(2)=\{0;4\}$
	\task $f^{-1}(0)=\{2\}$
	\task $z_f=\{2\}$
\end{tasks}
}
