\contenu{
	\tcblower
	Déterminer la préimage de $-1$, $0$, $3$, $\dfrac{2}{5}$ par les fonctions suivantes : 
	\begin{tasks}(4)
		\task $f(x)=3x+1$
		\task $g(x)=x^2-1$
		\task $h(x)=x^2$
		\task $i(x)=\sqrt{x}$
	\end{tasks}
}
\correction{
	\tcblower
\begin{tasks}(1)
	\task $f^{-1}(-1)=\left\{-\dfrac{2}{3}\right\}, f^{-1}(0)=\left\{-\dfrac{1}{3}\right\}, f^{-1}(3)=\left\{\dfrac{2}{3}\right\}, f^{-1}\left(\dfrac{2}{5}\right)=\left\{-\dfrac{1}{5}\right\}$
	\task $g^{-1}(-1)=\left\{0\right\}, g^{-1}(0)=\left\{-1;1\right\}, g^{-1}(3)=\left\{-2;2\right\}, g^{-1}\left(\dfrac{2}{5}\right)=\left\{-\sqrt{\dfrac{7}{5}};\sqrt{\dfrac{7}{5}}\right\}$
	\task $h^{-1}(-1)=\emptyset, h^{-1}(0)=\left\{0\right\}, h^{-1}(3)=\left\{-\sqrt{3};\sqrt{3}\right\}, h^{-1}\left(\dfrac{2}{5}\right)=\left\{-\sqrt{\dfrac{2}{5}};\sqrt{\dfrac{2}{5}}\right\}$
	\task $i^{-1}(-1)=\emptyset, i^{-1}(0)=\left\{0\right\}, i^{-1}(3)=\left\{9\right\}, i^{-1}\left(\dfrac{2}{5}\right)=\left\{\dfrac{4}{25}\right\}$
\end{tasks}
}
