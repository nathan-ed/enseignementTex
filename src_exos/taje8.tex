\task \(\displaystyle\lim_{x\to 2^-}f(x)=-\infty\)
\task \(\displaystyle\lim_{x\to +\infty}f(x)=+\infty\) et

\(\displaystyle\lim_{x\to -\infty}f(x)=1\)
\task \(\displaystyle\lim_{x\to 1}f(x)=4\)
\end{tasks}
