\correction{
	\tcblower

\begin{tasks}
\task \(y=(x-2)(2x-5)\).
On développe~: \(y=2x^2-9x+10\).
\[
y'=4x-9.
\]

\task \(y=\dfrac{1}{2}x^3+4x-\dfrac{3}{4}\).
\[
y'=\dfrac{3}{2}x^2+4.
\]

\task \(y=(x-1)^2(3x+2)\).
On développe~: \(y=(x^2-2x+1)(3x+2)=3x^3-4x^2-x+2\).
\[
y'=9x^2-8x-1.
\]

\task \(y=\dfrac{4}{x^3}=4x^{-3}\).
\[
y'=-12x^{-4}=-\dfrac{12}{x^4}.
\]

\task \(y=\dfrac{2}{x}-\dfrac{6}{2x^2}=2x^{-1}-3x^{-2}\).
\[
y'=-2x^{-2}+6x^{-3}=-\dfrac{2}{x^2}+\dfrac{6}{x^3}.
\]

\task \(y=\dfrac{7x-2}{2-7x}\).
En utilisant la règle du quotient~:
\[
y'=\dfrac{7(2-7x)-(-7)(7x-2)}{(2-7x)^2}=\dfrac{14-49x+49x-14}{(2-7x)^2}=0.
\]

\task \(y=\dfrac{x^2+x+1}{5x-3}\).
En utilisant la règle du quotient~:
\[
y'=\dfrac{(2x+1)(5x-3)-5(x^2+x+1)}{(5x-3)^2}=\dfrac{5x^2-6x-8}{(5x-3)^2}.
\]

\task \(y=x+3+\dfrac{x+3}{6x-3}\).
On réécrit~: \(y=(x+3)\left(1+\dfrac{1}{6x-3}\right)=(x+3)\cdot\dfrac{6x-2}{6x-3}\).
En utilisant la règle du produit et du quotient~:
\[
y'=\dfrac{6x-2}{6x-3}+(x+3)\cdot\dfrac{6(6x-3)-6(6x-2)}{(6x-3)^2}=\dfrac{6x-2}{6x-3}-\dfrac{6(x+3)}{(6x-3)^2}.
\]

\task \(y=\dfrac{16}{11x^2-8x+21}=16(11x^2-8x+21)^{-1}\).
En utilisant la règle de dérivation composée~:
\[
y'=-16(11x^2-8x+21)^{-2}(22x-8)=-\dfrac{16(22x-8)}{(11x^2-8x+21)^2}.
\]
\end{tasks}

}
