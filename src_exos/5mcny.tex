\contenu{
\tcblower
Dériver les fonctions en $x$ suivantes, à l'aide des propriétés de la dérivée ($a, b, c, d$ et $\pi$ sont des nombres réels) :
\begin{tasks}(3)
\task $f(x)=2 x-3$
\task $f(x)=\pi x^2$
\task $f(x)=4 x^2-5 x+6$
\task $f(x)=a x^2+b x+c$
\task $f(x)=(2 x-3)^2$
\task $f(x)=2 x-\frac{1}{x^2-1}$
\task $f(x)=\frac{x+5}{x-1}$
\task $f(x)=\frac{a}{x^2}$
\task $f(x)=\frac{1}{x^4}$
\task*(2) $f(x)=(2 x-1)(3-4 x)$
\task $f(x)=\frac{10}{x^3-4 x^2-2}$
\task $f(x)=\frac{a x+b}{c x+d}$
\end{tasks}
}
\correction{
\tcblower
\begin{tasks}(2)
\task $f'(x) = 2$

\task $f'(x) = 2\pi x$

\task $f'(x) = 8x - 5$

\task $f'(x) = 2ax + b$

\task $f'(x) = (2x-3)' \cdot 2(2x-3) = 2 \cdot 2(2x-3) = 8x - 12$

\task $f'(x) = 2 + \dfrac{(x^2-1)'}{(x^2-1)^2} = 2 + \dfrac{2x}{(x^2-1)^2}$

\task* $f'(x) = \dfrac{1 \cdot (x-1) - (x+5) \cdot 1}{(x-1)^2} = \dfrac{x - 1 - x - 5}{(x-1)^2} = \dfrac{-6}{(x-1)^2}$

\task $f'(x) = -\dfrac{2a}{x^3}$

\task $f'(x) = -\dfrac{4}{x^5}$

\task* $f'(x) = 2(3-4x) + (2x-1)(-4) = 6 - 8x - 8x + 4 = 10 - 16x$

\task* $f'(x) = -\dfrac{10(3x^2 - 8x)}{(x^3 - 4x^2 - 2)^2}$

\task* $f'(x) = \dfrac{a(cx+d) - (ax+b)c}{(cx+d)^2} = \dfrac{acx + ad - acx - bc}{(cx+d)^2} = \dfrac{ad - bc}{(cx+d)^2}$
\end{tasks}
}
