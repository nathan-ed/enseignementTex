On considère les fonctions données par : $f(x)=\sqrt{x}$, $g(x)=\sin (x)$ et $h(x)=4 x^3+3$.
\begin{tasks}	
\task Donner l'image de $x$ pour les fonctions suivantes :
\begin{multicols}{2}
\begin{enumerate}
\item $(f \circ h)(x)$;
\item $(g \circ f)(x)$;
\item $(g \circ h)(x)$;
\item $(h \circ g)(x)$;
\item $(f \circ g \circ h)(x)$
\end{enumerate}
\end{multicols}
\task Calculer la dérivée des fonctions $f \circ h, g \circ f, g \circ h, h \circ g$ et $f \circ g \circ h$.
\end{tasks}
}
