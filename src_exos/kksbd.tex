\textbf{Question 1 : $f(x) = \dfrac{x(3x + 4)}{2(x - 1)^2}$}

\textit{Domaine :} $\mathbb{R} \setminus \{1\}$

\textit{Dérivée première :}
\begin{align*}
f'(x) &= \frac{(6x + 4) \cdot 2(x - 1)^2 - x(3x + 4) \cdot 4(x - 1)}{4(x - 1)^4} \\
&= \frac{2(6x + 4)(x - 1) - 4x(3x + 4)}{4(x - 1)^3} \\
&= \frac{2(6x^2 - 2x - 4) - 4(3x^2 + 4x)}{4(x - 1)^3} \\
&= \frac{-12x - 8}{4(x - 1)^3} = \frac{-4(3x + 2)}{4(x - 1)^3}
\end{align*}

Point critique : $x = -\frac{2}{3}$

\textit{Dérivée seconde :} Calcul par la règle du quotient appliquée à $f'(x)$.

\textbf{Question 2 : $f(x) = \dfrac{1}{x^2 + 1}$}

\textit{Domaine :} $\mathbb{R}$

\textit{Dérivée première :}
\[
f'(x) = \frac{-2x}{(x^2 + 1)^2}
\]

Point critique : $x = 0$ (maximum car $f'$ change de signe de $+$ à $-$)

\textit{Dérivée seconde :}
\[
f''(x) = \frac{-2(x^2 + 1)^2 + 2x \cdot 2(x^2 + 1) \cdot 2x}{(x^2 + 1)^4} = \frac{6x^2 - 2}{(x^2 + 1)^3}
\]

Points d'inflexion : $x = \pm \frac{1}{\sqrt{3}}$

\textbf{Question 3 : $f(x) = \dfrac{x^2 - 1}{x^3} = \dfrac{1}{x} - \dfrac{1}{x^3}$}

\textit{Domaine :} $\mathbb{R}^*$

\textit{Dérivée première :}
\[
f'(x) = -\frac{1}{x^2} + \frac{3}{x^4} = \frac{3 - x^2}{x^4}
\]

\textit{Dérivée seconde :}

f''(x) = \frac{2}{x^3} - \frac{12}{x^5} = \frac{2x^2 - 12}{x^5}
\]

\textbf{Question 4 : $f(x) = \dfrac{x^5}{20} + \dfrac{x^4}{4}$}

\textit{Dérivée première :}
\[
f'(x) = \frac{x^4}{4} + x^3 = x^3\left(\frac{x}{4} + 1\right)
\]

Points critiques : $x = 0$ et $x = -4$

\textbf{Question 5 : $f(x) = 2 - \dfrac{1}{x + 1} + \dfrac{4}{x - 2}$}

\textit{Domaine :} $\mathbb{R} \setminus \{-1, 2\}$

\[
f'(x) = \frac{1}{(x + 1)^2} - \frac{4}{(x - 2)^2}
\]

$f'(x) = 0 \Leftrightarrow (x - 2)^2 = 4(x + 1)^2$

\textbf{Question 6 : $f(x) = x^5 - 3x^3 - 9x + 10$}

\[
f'(x) = 5x^4 - 9x^2 - 9 = 5x^4 - 9(x^2 + 1)
\]

\[
f''(x) = 20x^3 - 18x = 2x(10x^2 - 9)
\]

Points d'inflexion : $x = 0, \pm \frac{3}{\sqrt{10}}$
