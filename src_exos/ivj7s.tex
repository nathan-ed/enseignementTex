\begin{tasks}
	\task Pour chaque $a\in \{0;1,5;3;4;6;8\}$, tracer la tangente au graphe de $f$ en son point d'abscisse $a$ puis estimer $f(a)$ et $f'(a)$. 
	\task Estimer graphiquement les solutions des équations suivantes.
	\begin{multicols}{3}
		\begin{enumerate}
			\item $f(x)=0$
			\item $f'(x)=0$
			\item $f(x)=1$
			\item $f'(x)=1$
			\item $f(x)=-1$
			\item $f'(x)=-1$
		\end{enumerate}
	\end{multicols}
\end{tasks}
