\contenu{
\tcblower
	Trouver dans chaque intervalle: $\interval[open]{-4}{-3}\,;\,\interval[open]{\dfrac{1}{4}}{\dfrac{1}{3}}\,;\,\interval[open]{10^{-4}}{10^{-3}}$\,:
\begin{tasks}
\task deux nombres rationnels, l'un à partie décimale finie et l'autre à partie décimale infinie périodique 

(les donner sous forme de fraction irréductible);
\task un nombre irrationnel.
\end{tasks}
}
\correction{
\tcblower
Il y a une infinité de possibilités.
\begin{tasks}
	\task $-\dfrac{7}{5},-\dfrac{10}{3} \in \interval[open]{-4}{-3}$, $\dfrac{10}{3},\dfrac{27}{99}\in \interval[open]{\dfrac{1}{4}}{\dfrac{1}{3}}$, $\dfrac{5}{1000},\dfrac{1}{9000}\in \interval[open]{10^{-4}}{10^{-3}}$
	\task $-2,5\sqrt{2}$, $\dfrac{2}{5\sqrt{2}}$, $\dfrac{\sqrt{2}}{1000}$.
\end{tasks}
}
