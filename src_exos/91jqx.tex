\tcblower
Déterminer les valeurs de $a$ pour que les graphes des fonctions $f$ et $g$ se coupent à angle droit.
\begin{tasks}
\task $f(x)=x^2$ et $g(x)=1-a x^2$
\task $f(x)=a x^2$ et $g(x)=\frac{1-x^2}{a}$
\task $f(x)=2 x^2-a$ et $g(x)=\frac{x^2}{a}$
\end{tasks}
}
\correction{
\tcblower
\begin{tasks}(3)
\task $a=\frac{1}{3}$
\task $a= \pm \sqrt{3}$
\task impossible
\end{tasks}
}
