Soit $O$ le centre du cercle $(C)$. 

Déterminer la valeur des angles 
\[\widehat{AOB}; \quad \widehat{ABO};\quad \widehat{BAO}.\]
}

\begin{minipage}[t]{0.4\textwidth}{
\vspace{0pt}
\begin{center}
\begin{tikzpicture}[scale=0.6]
	\tkzSetUpPoint[shape=cross]
% Define the center and a point on the circle
  \tkzDefPoint(0,0){O}
  \tkzDefPoint(3,0){C}
  
\tkzLabelCircle[above right](O,C)(40){$(C)$}
  % Draw the circle with center O passing through A
  \tkzDrawCircle(O,C)
    % Compute the other vertices by successive rotations of 72°

  \tkzDefPointBy[rotation= center O angle 220](C) \tkzGetPoint{A}
  \tkzDefPointBy[rotation= center O angle 280](C) \tkzGetPoint{B}
  
  
  \tkzDrawPoints(A,B,C,O)
  
  \tkzDrawSegment(A,C)
  \tkzDrawSegment(B,A)
  \tkzDrawSegment(C,B)
  \tkzDrawSegment(O,B)
  \tkzDrawSegment(O,A)
  % Label the vertices
  \tkzLabelPoints[below left](A)
  \tkzLabelPoints[below right](B)
  \tkzLabelPoints[right](C)
  \tkzLabelPoints[above](O)
  \tkzMarkAngle[size=0.8cm, mark=none](A,C,B)
  \tkzLabelAngle[pos=1.5](A,C,B){\footnotesize{$25^\circ$}}
