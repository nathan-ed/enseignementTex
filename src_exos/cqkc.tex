$
  \left\{
    \begin{aligned}
      & \dfrac{x+y}{2} = \dfrac{7x - 5y}{6} + \dfrac{x+4}{4} \\

    \end{aligned}
  \right.
$

\task
$
  \left\{

  \right.
$

    \begin{aligned}

  \right.
$
