Soient $f(x) = 2x + 1$, $g(x) = x^2 - 4$, $h(x)=5$ et $k(x)=\sqrt{x}$
\begin{tasks}
\task Déterminer $(f + g)(x)$.
\task Déterminer $(f \cdot g)(x)$ sous forme développée.
\task Calculer la valeur exacte de l'image de $\sqrt{3}$ par $(f - g)$.
\task Déterminer $(g / f)(x)$ et préciser le domaine de définition.
\task Déterminer $(f \cdot h)(x)$.
\task Déterminer $(2f + 3g)(x)$.
\task Donner le domaine de définition de $(g + k)(x)$.
\task Montrer que $(f \cdot f)(x)$ est un trinôme du second degré et donner son expression sous la forme $ax^{2}+bx+c$.
\task Simplifier $(f^2 - g)(x)$.
\task Déterminer $i(x)$ telle que $(f + i)(x) = x^2 + 5x - 2$.
\end{tasks}
