\begin{tasks}(2)
\task $f(x) = \dfrac{2x^2 + 1}{x - 1}$

\textbf{Domaine :} $\mathbb{R} \setminus \{1\}$

\textbf{Asymptote verticale :} $x = 1$ (car $\lim_{x \to 1} f(x) = \pm\infty$)

\textbf{Asymptote oblique :} Division polynomiale : $2x^2 + 1 = (x - 1)(2x + 2) + 3$

Donc $f(x) = 2x + 2 + \dfrac{3}{x - 1}$, asymptote oblique $y = 2x + 2$

\textbf{Dérivée :} $f'(x) = \dfrac{4x(x - 1) - (2x^2 + 1)}{(x - 1)^2} = \dfrac{2x^2 - 4x - 1}{(x - 1)^2}$

Signe de $f'$ : $2x^2 - 4x - 1 = 0 \Rightarrow x = \dfrac{4 \pm \sqrt{16 + 8}}{4} = \dfrac{4 \pm 2\sqrt{6}}{4} = 1 \pm \dfrac{\sqrt{6}}{2}$

\task $f(x) = \dfrac{x^3 - 1}{3x^2}$

\textbf{Domaine :} $\mathbb{R}^*$

\textbf{Asymptote verticale :} $x = 0$

\textbf{Asymptote oblique :} $f(x) = \dfrac{x^3}{3x^2} - \dfrac{1}{3x^2} = \dfrac{x}{3} - \dfrac{1}{3x^2}$, asymptote oblique $y = \dfrac{x}{3}$

\textbf{Dérivée :} $f'(x) = \dfrac{3x^2 \cdot 3x^2 - (x^3 - 1) \cdot 6x}{9x^4} = \dfrac{9x^4 - 6x^4 + 6x}{9x^4} = \dfrac{3x^4 + 6x}{9x^4} = \dfrac{x^3 + 2}{3x^3}$

$f'(x) = 0$ si $x^3 = -2 \Rightarrow x = -\sqrt[3]{2}$

\task $f(x) = \dfrac{x^3}{x^2 - 1}$

\textbf{Domaine :} $\mathbb{R} \setminus \{-1, 1\}$

\textbf{Asymptotes verticales :} $x = -1$ et $x = 1$

\textbf{Asymptote oblique :} $f(x) = x + \dfrac{x}{x^2 - 1}$, asymptote oblique $y = x$

\textbf{Dérivée :} $f'(x) = \dfrac{3x^2(x^2 - 1) - x^3 \cdot 2x}{(x^2 - 1)^2} = \dfrac{3x^4 - 3x^2 - 2x^4}{(x^2 - 1)^2} = \dfrac{x^4 - 3x^2}{(x^2 - 1)^2} = \dfrac{x^2(x^2 - 3)}{(x^2 - 1)^2}$

$f'(x) = 0$ si $x = 0$ ou $x = \pm\sqrt{3}$
\end{tasks}
